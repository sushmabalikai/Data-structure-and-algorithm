void insert_end(int ele)
{
        tmp=p;
        tmp1=(struct node*)malloc(sizeof(struct node));
        tmp1->data=ele;
        tmp1->next=NULL;
        if(p==NULL)
        p=tmp1;
        else
        {
           while(tmp->next!=NULL)
            tmp=tmp->next;
            tmp->next=tmp1;
        }
 }
 void insert_beg(int ele)
 {
        tmp=(struct node*)malloc(sizeof(struct node));
        tmp->data=ele;
              if(p==NULL)
              {
              tmp->next=NULL;
              }
             else
              {
             tmp->next=p;
              }
           p=tmp;
}   
void delete_beg()
{
        tmp=p;
        if(p==NULL)
        printf("\n no element to be deleted!!");
        else
        {
            printf("\n element deleted=%d",p->data);
            p=p->next;
        }
}
void delete_end()
{
        tmp=p;
        struct node *pre;
        if(p==NULL)
        printf("\n no element to be deleted!!");
        else if(p->next==NULL)
        printf("\nelement deleted-%d",p->data);
        p=NULL;
    }
    else
    {
        while(tmp->next!=NULL)
        {
            pre=tmp;
            tmp=tmp->next;
        }
        pre->next=NULL;
        printf("\nelement deleted-%d",tmp->data);
    }
void display()
{
       tmp=p;
       while(tmp!=NULL)
        {
            printf("\n%d",tmp->data);
            tmp=tmp->next;
        }
}
